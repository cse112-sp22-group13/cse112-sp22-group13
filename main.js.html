function bindEnterOnMain () {
    document.addEventListener("keydown", function (event) {
        if (event.key === "Enter") {
            const element = document.activeElement;
            if (element.className === "search-bar") {
                window.location.href = "../recipe_list/recipe_list.html";
            }
        }
    });
}
